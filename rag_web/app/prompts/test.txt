ALGORITHM Generate_Visuals(ContentBlocks, MetadataContext)

BEGIN
    Placeholders ← Extract_Visual_Placeholders(ContentBlocks)

    PARALLEL FOR EACH Placeholder IN Placeholders DO

        VisualPlan ← LLM_Create_Visual_Plan(Placeholder)

        RETRY_COUNT ← 0

        WHILE RETRY_COUNT < 3 DO
            TRY
                SQLQuery ← Generate_SQL(VisualPlan)
                Data ← Execute_SQL(SQLQuery)

                Visual ← Generate_Chart(Data)

                Store(Visual)
                BREAK

            CATCH Error
                RETRY_COUNT ← RETRY_COUNT + 1
                VisualPlan ← Refine_VisualPlan(VisualPlan)
            ENDTRY
        ENDWHILE

    END PARALLEL FOR

RETURN AllVisuals
END
